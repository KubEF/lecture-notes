\section*{Производные функций}
\noindent\begin{gather*}
    (x^p)' = px^{p-1}\\
    (kx+b)' = k\\
    (c)'=0\\
    (e^x)'=e^x\\
    (e^{kx+b})'=ke^{kx+b}\\
    (a^x)'=a^x \ln a\\
    (\ln x)'= \frac{1}{x}, x>0\\
    (\ln (kx+b))'= \frac{k}{kx+b}\\
    (\log_a x)'=\frac{1}{x \ln a}\\
    (\sin x)'=\cos x\\
    (\cos x)' = -\sin x\\
    (\sin (kx+b))'=k\cos (kx+b)\\
    (\cos (kx+b))' = -k\sin (kx+b)\\
    (\tg x)' = \frac{1}{\cos^2 x}\\
    (\ctg x)' = -\frac{1}{\sin^2 x}
\end{gather*}
